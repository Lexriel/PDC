Qu'affiche le programme suivant~:
\begin{verbatim}
#include<stdio.h>
#define PRECEDENTE 1
#define ACTUELLE 2
#define PROCHAINE 3
int
main
(void) 
{
  int annee = ACTUELLE;
  switch (annee) 
  {
   case PRECEDENTE: annee = 2007;
   case ACTUELLE: annee = 2008;
   case PROCHAINE: annee = 2009;
   default:annee = 2010;
  }
  printf("Bonne annee %d\n", annee);
  return 0;
}
\end{verbatim}
\ifcorrection
\begin{correction}
Ce programme affiche
\begin{verbatim}
% ./a.out 
Bonne annee 2010
% 
\end{verbatim}
En effet, aucun \verb+break+ n'est utilis\'e dans le \verb+switch+.
Donc, toutes les instructions apr\`es \verb+ACTUELLE+ sont ex\'ecut\'ees.
\end{correction}
\fi
